Expose the Phase 1 cross-poll failure tracker through the existing Hono API so the dashboard UI (Plans 02 and 03) can read and mutate it.

Purpose: The tracker already lives in `bots.casCacheJson.dateFailureTracking` (Phase 1, deployed 2026-04-07). This plan is the thin backend slice that unblocks both UI plans.

export type FailureDimension = 'consularNoTimes' | 'consularNoDays' | 'casNoTimes' | 'casNoDays';

export interface DateFailureEntry {
  windowStartedAt: string;       // ISO 8601 Z
  totalCount: number;
  byDimension: Partial<Record<FailureDimension, number>>;
  lastFailureAt: string;         // ISO 8601 Z
  blockedUntil?: string;         // ISO 8601 Z — present only when totalCount >= 5
}

export interface CasCacheData {
  refreshedAt: string;

  totalDates: number;
  fullDates: number;
  entries: CasCacheEntry[];
  blockedConsularDates?: Record<string, string>;
  dateFailureTracking?: Record<string, DateFailureEntry>;
}

       // Manual tracker unblock — single date
       botsRouter.delete('/:id/tracker/:date', async (c) => {
         const id = parseInt(c.req.param('id'));
         const date = c.req.param('date');
         const [bot] = await db.select({ casCacheJson: bots.casCacheJson }).from(bots).where(eq(bots.id, id));
         if (!bot) return c.json({ error: 'Bot not found' }, 404);
         const cache = bot.casCacheJson as CasCacheData | null;
         const tracking = cache?.dateFailureTracking ?? {};
         if (!tracking[date]) return c.json({ error: 'Date not in tracker' }, 404);
         await db.update(bots).set({
           casCacheJson: sql`jsonb_set(
             coalesce(${bots.casCacheJson}, '{}'::jsonb),
             '{dateFailureTracking}',
             coalesce(${bots.casCacheJson}->'dateFailureTracking', '{}'::jsonb) - ${date}
           )`,
           updatedAt: new Date(),
         }).where(eq(bots.id, id));
         return c.json({ ok: true });
       });

       // Manual tracker clear — all entries
       botsRouter.delete('/:id/tracker', async (c) => {
         const id = parseInt(c.req.param('id'));
         const [bot] = await db.select({ casCacheJson: bots.casCacheJson }).from(bots).where(eq(bots.id, id));
         if (!bot) return c.json({ error: 'Bot not found' }, 404);
         const cache = bot.casCacheJson as CasCacheData | null;
         const cleared = Object.keys(cache?.dateFailureTracking ?? {}).length;
         await db.update(bots).set({
